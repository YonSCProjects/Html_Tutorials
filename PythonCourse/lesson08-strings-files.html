תרגיל 1.2: טבלת שיטות מחרוזת

שיטהמה עושהדוגמה
.upper()הופך לאותיות גדולות"hi".upper() → "HI"
.lower()הופך לאותיות קטנות"HI".lower() → "hi"
.strip()מסיר רווחים מהקצוות" hi ".strip() → "hi"
.split(x)מפצל לרשימה"a,b".split(",") → ['a','b']
.join(list)מחבר רשימה למחרוזת"-".join(['a','b']) → "a-b"
.replace(a,b)מחליף טקסט"hi".replace("h","b") → "bi"
.find(x)מוצא מיקום (או -1)"hello".find("ll") → 2
.count(x)סופר הופעות"aaba".count("a") → 3

תרגיל 1.3: צופן קיסר

  1. בנה מקודד ומפענח צופן קיסר:
def caesar_encrypt(text, shift):
    result = ""
    for char in text:
        if char.isalpha():
            base = ord('A') if char.isupper() else ord('a')
            shifted = (ord(char) - base + shift) % 26 + base
            result += chr(shifted)
        else:
            result += char
    return result

def caesar_decrypt(text, shift):
    return caesar_encrypt(text, -shift)

secret = caesar_encrypt("Hello World", 3)
print(f"מוצפן: {secret}")
print(f"מפוענח: {caesar_decrypt(secret, 3)}")

מוצפן: Khoor Zruog
מפוענח: Hello World

תרגיל 2.1: f-strings — הדרך המודרנית

  1. f-strings הם הדרך הכי נוחה לפרמט מחרוזות:

# דרך 4: לולאה על שורות (הכי מומלץ!)

    for i, line in enumerate(f, 1):

למה with? הפקודה with סוגרת את הקובץ אוטומטית כשמסיימים, גם אם קרתה שגיאה. תמיד השתמשו ב-with!

def count_words(filename):
    with open(filename, "r", encoding="utf-8") as f:
        content = f.read()

    words = content.split()
    lines = content.count("\n")
    chars = len(content)

    print(f"מילים: {len(words)}")
    print(f"שורות: {lines}")
    print(f"תווים: {chars}")

count_words("sample.txt")

תרגיל 5.1: אפליקציית יומן מלאה

  1. צור קובץ diary.py והעתק את הקוד:
from datetime import datetime

DIARY_FILE = "my_diary.txt"

def add_entry():
    now = datetime.now().strftime("%Y-%m-%d %H:%M")
    print("כתוב את הרשומה (Enter לסיום):")
    entry = input("> ")

    with open(DIARY_FILE, "a", encoding="utf-8") as f:
        f.write(f"\n--- {now} ---\n")
        f.write(entry + "\n")

    print("הרשומה נשמרה!")

def read_entries():
    try:
        with open(DIARY_FILE, "r", encoding="utf-8") as f:
            content = f.read()
        if content.strip():
            print("\n📖 היומן שלך:")
            print(content)
        else:
            print("היומן ריק.")
    except FileNotFoundError:
        print("אין עדיין יומן. כתוב רשומה ראשונה!")

def search_entries(keyword):
    try:
        with open(DIARY_FILE, "r", encoding="utf-8") as f:
            lines = f.readlines()

        found = False
        for i, line in enumerate(lines):
            if keyword in line:
                print(f"שורה {i+1}: {line.strip()}")
                found = True

        if not found:
            print(f"לא נמצא '{keyword}' ביומן.")
    except FileNotFoundError:
        print("אין עדיין יומן.")

# תפריט ראשי
while True:
    print("\n=== היומן שלי ===")
    print("1. כתוב רשומה חדשה")
    print("2. קרא את היומן")
    print("3. חפש ביומן")
    print("4. יציאה")

    choice = input("בחר: ")

    if choice == "1":
        add_entry()
    elif choice == "2":
        read_entries()
    elif choice == "3":
        keyword = input("מילת חיפוש: ")
        search_entries(keyword)
    elif choice == "4":
        print("להתראות!")
        break
    else:
        print("בחירה לא תקינה!")
