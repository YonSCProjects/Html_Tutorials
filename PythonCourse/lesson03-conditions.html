תרגיל 3.1: if / elif / else

כשיש כמה אפשרויות, משתמשים ב-elif (קיצור של "else if"):

grade = int(input("מה הציון שלך? "))

if grade >= 90:
    print("מצוין! 🌟")
elif grade >= 80:
    print("טוב מאוד! 👍")

    print("טוב 😊")

    print("מספיק 🤷")
else:
    print("צריך לשפר... 📚")
מה הציון שלך? 85
טוב מאוד! 👍
Python בודק את התנאים מלמעלה למטה ונכנס רק לענף הראשון שמתאים. else תופס את כל מה שלא נתפס לפני כן.

תרגיל 3.3: אופרטורים לוגיים — and, or, not

# and - שני התנאים חייבים להיות נכונים

has_ticket = True

if age >= 12 and has_ticket:
    print("אתה יכול להיכנס לסרט! 🎬")

# or - מספיק שתנאי אחד נכון
day = "שבת"
if day == "שישי" or day == "שבת":
    print("סוף שבוע! 🎉")

# not - הופך את התנאי
is_raining = False
if not is_raining:
    print("אפשר לצאת החוצה! ☀️")

תרגיל 3.4: מחשבון אינטראקטיבי

בואו נבנה מחשבון שמבקש מהמשתמש שני מספרים ופעולה:

num1 = float(input("מספר ראשון: "))
operation = input("פעולה (+, -, *, /): ")
num2 = float(input("מספר שני: "))

if operation == "+":
    result = num1 + num2
elif operation == "-":
    result = num1 - num2
elif operation == "*":
    result = num1 * num2
elif operation == "/":
    if num2 == 0:
        print("שגיאה: לא ניתן לחלק באפס! ❌")
        result = None
    else:
        result = num1 / num2
else:
    print("פעולה לא מוכרת! ❌")
    result = None

if result is not None:
    print(f"{num1} {operation} {num2} = {result}")
מספר ראשון: 15
פעולה (+, -, *, /): *
מספר שני: 4
15.0 * 4.0 = 60.0
✅ עצור! ודא שאתה מבין: if/elif/else, כל אופרטורי ההשוואה, ו-and/or/not.

is_long = len(password) >= 8
has_number = False
for char in password:
    if char.isdigit():
        has_number = True

if is_long and has_number:
    print("✅ סיסמה חזקה!")
elif is_long:
    print("⚠️ טוב, אבל הוסף מספרים")
elif has_number:
