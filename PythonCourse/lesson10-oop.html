משימה 1: מה זה OOP?

⏱️ 10 דקות

תכנות מונחה עצמים (Object-Oriented Programming) הוא דרך לארגן קוד לפי "דברים" מהעולם האמיתי.

אנלוגיה: חשוב על מחלקה (class) כמו תוכנית בנייה של בית. התוכנית עצמה היא לא בית — אבל ממנה אפשר לבנות הרבה בתים. כל בית שנבנה מהתוכנית הוא אובייקט (object).

class BankAccount:
    def __init__(self, owner, balance=0):
        self.owner = owner
        self.balance = balance

    def deposit(self, amount):
        self.balance += amount
        print(f"Deposited {amount}. Balance: {self.balance}")

    def withdraw(self, amount):
        if amount > self.balance:
            print("Not enough funds!")
        else:
            self.balance -= amount
            print(f"Withdrew {amount}. Balance: {self.balance}")

# Test it
acc = BankAccount("Yoni", 1000)
acc.deposit(500)
acc.withdraw(200)
acc.withdraw(2000)

Withdrew 200. Balance: 1300
Not enough funds!

אם נתחיל עם balance=1000, נפקיד 500, ונמשוך 200 — מה היתרה הסופית?

תרגיל 3.1: __str__ להדפסה יפה

  1. צור קובץ student_class.py
  2. כתוב:
class Student:
    def __init__(self, name, grades=None):
        self.name = name
        self.grades = grades if grades else []

    def add_grade(self, grade):
        self.grades.append(grade)

    def average(self):
        if not self.grades:
            return 0
        return sum(self.grades) / len(self.grades)

    def __str__(self):
        return f"Student {self.name}, avg: {self.average():.1f}"

    def __repr__(self):
        return f"Student('{self.name}', {self.grades})"

s = Student("Dana")
s.add_grade(90)
s.add_grade(85)
s.add_grade(95)
print(s)          # uses __str__
print(repr(s))   # uses __repr__
Student Dana, avg: 90.0
Student('Dana', [90, 85, 95])

מה ההבדל בין __str__ ל-__repr__?

class Pet:
    def __init__(self, name, pet_type):
        self.name = name
        self.pet_type = pet_type
        self.hunger = 50    # 0 = full, 100 = starving
        self.happiness = 50  # 0 = sad, 100 = ecstatic
        self.energy = 50     # 0 = exhausted, 100 = hyper

    def feed(self):
        self.hunger = max(0, self.hunger - 20)
        self.happiness += 5
        print(f"🍖 {self.name} ate! Hunger: {self.hunger}")

    def play(self):
        self.happiness = min(100, self.happiness + 20)
        self.energy = max(0, self.energy - 15)
        self.hunger += 10
        print(f"🎾 {self.name} played! Happy: {self.happiness}")

    def sleep(self):
        self.energy = min(100, self.energy + 30)
        print(f"💤 {self.name} slept! Energy: {self.energy}")

    def __str__(self):
        return (f"🐾 {self.name} ({self.pet_type}) | "
                f"Hunger: {self.hunger} | Happy: {self.happiness} | "
                f"Energy: {self.energy}")

my_pet = Pet("Pixel", "Cat")
my_pet.feed()
my_pet.play()
my_pet.sleep()
print(my_pet)
🍖 Pixel ate! Hunger: 30
🎾 Pixel played! Happy: 75
💤 Pixel slept! Energy: 65
🐾 Pixel (Cat) | Hunger: 40 | Happy: 75 | Energy: 65

משימה 4: ירושה (Inheritance)

⏱️ 15 דקות
ירושה מאפשרת למחלקה חדשה (ילד) לקבל תכונות ומתודות ממחלקה קיימת (הורה). כמו שילד יורש תכונות מההורים שלו — אבל יכול גם להיות ייחודי!

תרגיל 4.1: הורה וילד

  1. צור קובץ inheritance.py
  2. כתוב:
class Animal:
    def __init__(self, name, sound):
        self.name = name
        self.sound = sound

    def speak(self):
        return f"{self.name} says {self.sound}!"

class Dog(Animal):  # Dog inherits from Animal
    def __init__(self, name, breed):
        super().__init__(name, "Woof")  # Call parent __init__
        self.breed = breed

    def fetch(self):
        return f"{self.name} fetches the ball!"

class Cat(Animal):
    def __init__(self, name, indoor=True):
        super().__init__(name, "Meow")
        self.indoor = indoor

    def speak(self):  # Override parent method
        return f"{self.name} purrs softly... {self.sound}"

dog = Dog("Rex", "Husky")
cat = Cat("Mimi")

print(dog.speak())    # Uses Animal's speak
print(dog.fetch())    # Dog-only method
print(cat.speak())    # Uses Cat's overridden speak
print(isinstance(dog, Animal))  # True!
Rex says Woof!
Rex fetches the ball!
Mimi purrs softly... Meow
True

מה הפלט של isinstance(dog, Animal)?

✅ עצור! ודא שהבנת: super() קורא למתודה של ההורה. דריסת מתודות (override) מאפשרת לשנות התנהגות.

class SecureAccount:
    def __init__(self, owner, password, balance=0):
        self.owner = owner          # public
        self._password = password   # "private" (convention)
        self.__balance = balance    # name-mangled (harder to access)

    def get_balance(self, password):
        if password == self._password:
            return self.__balance
        return "Wrong password!"

    def deposit(self, amount, password):
        if password == self._password:
            self.__balance += amount
            return f"New balance: {self.__balance}"
        return "Wrong password!"

acc = SecureAccount("Dan", "secret123", 1000)
print(acc.owner)                        # Works fine
print(acc.get_balance("secret123"))     # Correct password
print(acc.get_balance("wrong"))         # Wrong password

תרגיל 6.1: מחלקת הבסיס Character

  1. צור קובץ rpg_game.py
  2. בנה את מחלקת הבסיס:
import random

class Character:
    def __init__(self, name, health, attack, defense):
        self.name = name
        self.health = health
        self.max_health = health
        self.attack = attack
        self.defense = defense
        self.is_alive = True

    def take_damage(self, damage):
        actual_damage = max(1, damage - self.defense)
        self.health -= actual_damage
        if self.health <= 0:
            self.health = 0
            self.is_alive = False
        print(f"  {self.name} takes {actual_damage} damage! HP: {self.health}/{self.max_health}")

    def basic_attack(self, target):
        damage = random.randint(self.attack - 2, self.attack + 2)
        print(f"⚔️ {self.name} attacks {target.name}!")
        target.take_damage(damage)

    def __str__(self):
        status = "ALIVE" if self.is_alive else "DEAD"
        return f"{self.name} | HP: {self.health}/{self.max_health} | ATK: {self.attack} | DEF: {self.defense} | {status}"

תרגיל 6.2: מחלקות הילד

  1. הוסף את המחלקות הבאות לקובץ:
class Warrior(Character):
    def __init__(self, name):
        super().__init__(name, health=120, attack=15, defense=8)

    def shield_bash(self, target):
        damage = self.attack + 5
        print(f"🛡️ {self.name} SHIELD BASHES {target.name}!")
        target.take_damage(damage)

class Mage(Character):
    def __init__(self, name):
        super().__init__(name, health=80, attack=20, defense=3)
        self.mana = 100

    def fireball(self, target):
        if self.mana >= 25:
            self.mana -= 25
            damage = self.attack + 10
            print(f"🔥 {self.name} casts FIREBALL on {target.name}! (Mana: {self.mana})")
            target.take_damage(damage)
        else:
            print(f"💨 {self.name} has no mana!")
            self.basic_attack(target)

class Archer(Character):
    def __init__(self, name):
        super().__init__(name, health=90, attack=18, defense=5)
        self.arrows = 10

    def power_shot(self, target):
        if self.arrows > 0:
            self.arrows -= 1
            damage = self.attack + 8
            print(f"🏹 {self.name} fires POWER SHOT! (Arrows: {self.arrows})")
            target.take_damage(damage)
        else:
            print(f"🏹 {self.name} is out of arrows!")
            self.basic_attack(target)

תרגיל 6.3: מערכת קרב!

  1. הוסף את פונקציית הקרב:
def battle(char1, char2):
    print("=" * 40)
    print(f"⚔️  BATTLE: {char1.name} vs {char2.name}  ⚔️")
    print("=" * 40)
    print(char1)
    print(char2)
    print()

    turn = 1
    while char1.is_alive and char2.is_alive:
        print(f"--- Turn {turn} ---")

        # char1 attacks
        if isinstance(char1, Warrior) and random.random() > 0.5:
            char1.shield_bash(char2)
        elif isinstance(char1, Mage):
            char1.fireball(char2)
        elif isinstance(char1, Archer):
            char1.power_shot(char2)
        else:
            char1.basic_attack(char2)

        if not char2.is_alive:
            break

        # char2 attacks
        if isinstance(char2, Warrior) and random.random() > 0.5:
            char2.shield_bash(char1)
        elif isinstance(char2, Mage):
            char2.fireball(char1)
        elif isinstance(char2, Archer):
            char2.power_shot(char1)
        else:
            char2.basic_attack(char1)

        turn += 1
        print()

    winner = char1 if char1.is_alive else char2
    print(f"\n🏆 {winner.name} WINS!")

# Let's battle!
w = Warrior("Thorin")
m = Mage("Gandalf")
battle(w, m)
