grades = {"מתמטיקה": 95, "אנגלית": 88, "היסטוריה": 92}

print("מקצועות:")
for subject in grades.keys():
    print(f"  {subject}")

# 2. values() — רק הערכים
print("\nציונים:")
for grade in grades.values():
    print(f"  {grade}")

# 3. items() — מפתח וערך ביחד (הכי שימושי!)
print("\nגיליון ציונים:")
for subject, grade in grades.items():
    print(f"  {subject}: {grade}")

תרגיל 3.2: ממוצע ציונים

grades = {"מתמטיקה": 95, "אנגלית": 88, "היסטוריה": 92, "מדעים": 97}

total = sum(grades.values())
count = len(grades)
average = total / count
print(f"ממוצע: {average}")  # ממוצע: 93.0

# מצא את המקצוע עם הציון הגבוה ביותר
best = max(grades, key=grades.get)
print(f"הכי טוב: {best} ({grades[best]})")  # הכי טוב: מדעים (97)

תרגיל 3.3: מונה מילים

בואו נבנה מונה שסופר כמה פעמים כל מילה מופיעה:

text = "the cat sat on the mat the cat"
words = text.split()

counter = {}
for word in words:
    if word in counter:
        counter[word] += 1
    else:
        counter[word] = 1

print(counter)
# {'the': 3, 'cat': 2, 'sat': 1, 'on': 1, 'mat': 1}

# מי המילה הכי נפוצה?
most_common = max(counter, key=counter.get)
print(f"הכי נפוצה: '{most_common}' ({counter[most_common]} פעמים)")

תרגיל 4.1: מה זה set?

סט הוא אוסף של ערכים ייחודיים — אין כפילויות!

# יצירת סט
colors = {"אדום", "כחול", "ירוק", "אדום", "כחול"}
print(colors)  # {'אדום', 'כחול', 'ירוק'} — בלי כפילויות!

# הסרת כפילויות מרשימה
numbers = [1, 2, 2, 3, 3, 3, 4, 4, 4, 4]
unique = set(numbers)
print(unique)       # {1, 2, 3, 4}
print(len(unique))  # 4

# הוספה ומחיקה
colors.add("צהוב")
colors.discard("אדום")
print(colors)  # {'כחול', 'ירוק', 'צהוב'}

תרגיל 4.2: פעולות על סטים

סטים תומכים בפעולות מתמטיות מגניבות:

a = {1, 2, 3, 4, 5}
b = {4, 5, 6, 7, 8}

# איחוד — כל מה שבאחד מהם
print(a | b)       # {1, 2, 3, 4, 5, 6, 7, 8}
print(a.union(b))  # אותו דבר

# חיתוך — מה שמשותף לשניהם
print(a & b)             # {4, 5}
print(a.intersection(b)) # אותו דבר

# הפרש — מה שב-a אבל לא ב-b
print(a - b)             # {1, 2, 3}
print(a.difference(b))   # אותו דבר

contacts = {
    "אמא": "050-1234567",
    "אבא": "052-7654321",
    "סבתא": "03-9876543"
}

def show_menu():
    print("\n📱 ספר טלפונים")
    print("1. הצג כל אנשי הקשר")
    print("2. חפש איש קשר")
    print("3. הוסף איש קשר")
    print("4. מחק איש קשר")
    print("5. יציאה")

def show_all():
    if not contacts:
        print("ספר הטלפונים ריק!")
        return
    print(f"\n{'שם':<15} {'טלפון'}")
    print("-" * 30)
    for name, phone in contacts.items():
        print(f"{name:<15} {phone}")
    print(f"\nסה\"כ: {len(contacts)} אנשי קשר")

def search():
    name = input("חפש שם: ")
    if name in contacts:
        print(f"📞 {name}: {contacts[name]}")
    else:
        # חיפוש חלקי
        found = {k: v for k, v in contacts.items() if name in k}
        if found:
            print("תוצאות:")
            for k, v in found.items():
                print(f"  📞 {k}: {v}")
        else:
            print("לא נמצא 😕")

def add_contact():
    name = input("שם: ")
    phone = input("טלפון: ")
    contacts[name] = phone
    print(f"✅ נוסף: {name} — {phone}")

def delete_contact():
    name = input("שם למחיקה: ")
    if name in contacts:
        removed = contacts.pop(name)
        print(f"🗑️ נמחק: {name} ({removed})")
    else:
        print("לא נמצא!")

while True:
    show_menu()
    choice = input("\nבחר (1-5): ")
    if choice == "1": show_all()
    elif choice == "2": search()
    elif choice == "3": add_contact()
    elif choice == "4": delete_contact()
    elif choice == "5":
        print("להתראות! 👋")
        break
    else:
        print("בחירה לא חוקית")
