תרגיל 2.2: תפיסת שגיאות ספציפיות

def safe_divide(a, b):
    try:
        result = a / b
        return result
    except ZeroDivisionError:
        print("אי אפשר לחלק באפס!")
        return None
    except TypeError:
        print("הערכים חייבים להיות מספרים!")
        return None

print(safe_divide(10, 2))    # 5.0
print(safe_divide(10, 0))    # אי אפשר לחלק באפס! None
print(safe_divide("a", 2))   # הערכים חייבים להיות מספרים! None
כלל חשוב: תמיד תפוס שגיאות ספציפיות! הימנע מ-except: כללי כי הוא מסתיר באגים.

משימה 3: else ו-finally

⏱️ 8 דקות

תרגיל 3.1: המבנה המלא

def read_file_safe(filename):
    try:
        with open(filename, "r", encoding="utf-8") as f:
            content = f.read()
    except FileNotFoundError:
        print(f"הקובץ {filename} לא נמצא!")
    except PermissionError:
        print("אין הרשאה לקרוא את הקובץ!")
    else:
        # רץ רק אם לא הייתה שגיאה
        print(f"הקובץ נקרא בהצלחה! ({len(content)} תווים)")
        return content
    finally:
        # רץ תמיד — גם אם הייתה שגיאה!
        print("--- סיום ניסיון קריאה ---")

read_file_safe("sample.txt")   # הקובץ נקרא בהצלחה!
read_file_safe("missing.txt")  # הקובץ missing.txt לא נמצא!
סדר הביצוע: try → (אם שגיאה: except) → (אם הצלחה: else) → (תמיד: finally)

תרגיל 6.1: המחשבון המושלם

  1. צור קובץ safe_calc.py:
def get_number(prompt):
    """מבקש מספר מהמשתמש עד שמקבל קלט תקין"""
    while True:
        try:
            return float(input(prompt))
        except ValueError:
            print("  זה לא מספר חוקי! נסה שוב.")

def get_operator():
    """מבקש פעולה חשבונית תקינה"""
    valid = ['+', '-', '*', '/', '**', '%']
    while True:
        op = input(f"פעולה ({', '.join(valid)}): ")
        if op in valid:
            return op
        print(f"  פעולה לא תקינה! בחר מתוך: {', '.join(valid)}")

def calculate(a, op, b):
    """מבצע חישוב עם טיפול בשגיאות"""
    try:
        if op == '+': return a + b
        elif op == '-': return a - b
        elif op == '*': return a * b
        elif op == '/':
            if b == 0:
                raise ZeroDivisionError("אי אפשר לחלק באפס!")
            return a / b
        elif op == '**': return a ** b
        elif op == '%': return a % b
    except OverflowError:
        print("  התוצאה גדולה מדי!")
        return None

def save_history(history):
    """שומר היסטוריה לקובץ"""
    try:
        with open("calc_history.txt", "w", encoding="utf-8") as f:
            for entry in history:
                f.write(entry + "\n")
        print(f"  נשמרו {len(history)} חישובים לקובץ.")
    except IOError as e:
        print(f"  שגיאה בשמירה: {e}")

# --- תוכנית ראשית ---
print("=== מחשבון חסין שגיאות ===")
print("הקלד 'exit' ליציאה, 'history' להיסטוריה\n")

history = []

while True:
    user_input = input("\n> חישוב חדש (או exit/history): ").strip()

    if user_input.lower() == 'exit':
        if history:
            save_history(history)
        print("להתראות!")
        break
    elif user_input.lower() == 'history':
        if history:
            print("\n📋 היסטוריה:")
            for i, h in enumerate(history, 1):
                print(f"  {i}. {h}")
        else:
            print("  אין היסטוריה עדיין.")
        continue

    a = get_number("מספר ראשון: ")
    op = get_operator()
    b = get_number("מספר שני: ")

    try:
        result = calculate(a, op, b)
        if result is not None:
            # עיגול אם צריך
            if result == int(result):
                result = int(result)
            entry = f"{a} {op} {b} = {result}"
            print(f"\n  ✅ {entry}")
            history.append(entry)
    except ZeroDivisionError as e:
        print(f"\n  ❌ {e}")
    except Exception as e:
        print(f"\n  ❌ שגיאה לא צפויה: {e}")
