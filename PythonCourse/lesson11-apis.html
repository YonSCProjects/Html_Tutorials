משימה 1: מה זה API?

⏱️ 8 דקות
אנלוגיה — המלצר במסעדה:
אתה (הלקוח) יושב במסעדה. אתה לא הולך ישירות למטבח — אתה מזמין דרך המלצר. המלצר לוקח את ההזמנה שלך למטבח (השרת) ומחזיר לך את האוכל (הנתונים). API = המלצר — ממשק שמאפשר לתוכנה שלך לדבר עם שרתים אחרים.

תרגיל 1.1: HTTP בסיסי

כשאתה גולש באינטרנט, הדפדפן שולח בקשות HTTP. שני הסוגים העיקריים:

# HTTP Methods - the two main ones:
#
# GET  = "Give me data"  (read)
#   Example: Open a webpage, search Google
#
# POST = "Here is data"  (write/create)
#   Example: Submit a form, upload a photo
#
# Status Codes:
#   200 = OK (success)
#   404 = Not Found
#   500 = Server Error

מה קוד הסטטוס שאומר "הכל בסדר"?

import requests

def safe_get(url):
    try:
        response = requests.get(url, timeout=5)
        response.raise_for_status()  # Raise error for 4xx/5xx
        return response
    except requests.exceptions.ConnectionError:
        print("Connection error! Check your internet.")
    except requests.exceptions.Timeout:
        print("Request timed out!")
    except requests.exceptions.HTTPError as e:
        print(f"HTTP Error: {e}")
    return None

# Test with a working URL
r = safe_get("https://catfact.ninja/fact")
if r:
    print("Success!", r.json()['fact'])

# Test with a broken URL
r = safe_get("https://catfact.ninja/nonexistent")
if not r:
    print("Request failed, but we handled it!")

import requests

# Method 1: Parameters in URL
response = requests.get("https://v2.jokeapi.dev/joke/Programming?type=single")

# Method 2: Parameters as dict (cleaner!)

    'lang': 'en'
}
response = requests.get("https://v2.jokeapi.dev/joke/Any", params=params)
data = response.json()

print("URL sent:", response.url)
print("Category:", data.get('category'))
print("Joke:", data.get('joke', 'No joke found'))

import requests

def get_weather(city):
    """Get weather for a city using wttr.in"""
    url = f"https://wttr.in/{city}?format=j1"

    try:
        response = requests.get(url, timeout=10)
        response.raise_for_status()
        data = response.json()

        # Extract current conditions
        current = data['current_condition'][0]

        weather = {
            'temp_c': current['temp_C'],
            'feels_like': current['FeelsLikeC'],
            'description': current['weatherDesc'][0]['value'],
            'humidity': current['humidity'],
            'wind_speed': current['windspeedKmph'],
        }
        return weather

    except requests.exceptions.RequestException as e:
        print(f"Error: {e}")
        return None

def display_weather(city, weather):
    if not weather:
        print(f"Could not get weather for {city}")
        return

    print("=" * 35)
    print(f"  Weather in {city}")
    print("=" * 35)
    print(f"  Temp:        {weather['temp_c']}°C")
    print(f"  Feels like:  {weather['feels_like']}°C")
    print(f"  Conditions:  {weather['description']}")
    print(f"  Humidity:    {weather['humidity']}%")
    print(f"  Wind:        {weather['wind_speed']} km/h")
    print("=" * 35)

# Test it!
cities = ["Tel+Aviv", "London", "Tokyo"]
for city in cities:
    weather = get_weather(city)
    display_weather(city, weather)
    print()

import requests

def get_joke():
    """Get a random joke from JokeAPI"""
    url = "https://v2.jokeapi.dev/joke/Any"
    params = {'type': 'single', 'lang': 'en'}
    response = requests.get(url, params=params)
    data = response.json()
    return data.get('joke', 'No joke found')

def get_cat_fact():
    """Get a random cat fact"""
    response = requests.get("https://catfact.ninja/fact")
    return response.json()['fact']

def get_number_fact(number):
    """Get a fun fact about a number"""
    response = requests.get(f"http://numbersapi.com/{number}?json")
    return response.json()['text']

# Run them all!
print("🃏 Random Joke:")
print("  ", get_joke())
print()

print("🐱 Cat Fact:")
print("  ", get_cat_fact())
print()

print("🔢 Fun fact about 7:")
print("  ", get_number_fact(7))
