import random
import string

def generate_password(length=12, use_upper=True, use_digits=True, use_symbols=True):
    """Generate a secure random password."""
    chars = string.ascii_lowercase  # always include lowercase

    if use_upper:
        chars += string.ascii_uppercase
    if use_digits:
        chars += string.digits
    if use_symbols:
        chars += string.punctuation

    # Ensure at least one of each required type
    password = []
    password.append(random.choice(string.ascii_lowercase))
    if use_upper:
        password.append(random.choice(string.ascii_uppercase))
    if use_digits:
        password.append(random.choice(string.digits))
    if use_symbols:
        password.append(random.choice(string.punctuation))

    # Fill the rest randomly
    while len(password) < length:
        password.append(random.choice(chars))

    # Shuffle so the required chars aren't always first
    random.shuffle(password)
    return ''.join(password)

import json
from datetime import datetime

def save_password(service, password, filename="passwords.json"):
    """Save password to a JSON file."""
    try:
        with open(filename, 'r') as f:
            data = json.load(f)
    except (FileNotFoundError, json.JSONDecodeError):
        data = []

    data.append({
        'service': service,
        'password': password,
        'created': datetime.now().strftime("%Y-%m-%d %H:%M")
    })

    with open(filename, 'w') as f:
        json.dump(data, f, indent=2)
    print(f"Password saved for {service}!")

def main():
    print("🔐 Password Generator")

import json
import os

class ContactManager:
    def __init__(self, filename="contacts.json"):
        self.filename = filename
        self.contacts = self._load()

    def _load(self):
        """Load contacts from file."""
        if os.path.exists(self.filename):
            with open(self.filename, 'r', encoding='utf-8') as f:
                return json.load(f)
        return {}

    def _save(self):
        """Save contacts to file."""
        with open(self.filename, 'w', encoding='utf-8') as f:
            json.dump(self.contacts, f, indent=2, ensure_ascii=False)

    def add(self, name, phone, email="", category="general"):
        """Add a new contact."""
        self.contacts[name] = {
            'phone': phone,
            'email': email,
            'category': category
        }
        self._save()
        print(f"Added: {name}")

    def search(self, query):
        """Search contacts by name (partial match)."""
        results = {}
        for name, info in self.contacts.items():
            if query.lower() in name.lower():
                results[name] = info
        return results

    def delete(self, name):
        """Delete a contact."""
        if name in self.contacts:
            del self.contacts[name]
            self._save()
            print(f"Deleted: {name}")
        else:
            print(f"Not found: {name}")

    def list_all(self):
        """Display all contacts."""
        if not self.contacts:
            print("  No contacts yet.")
            return
        for name, info in sorted(self.contacts.items()):
            print(f"  {name}: {info['phone']} | {info['email']} [{info['category']}]")

    def list_by_category(self, category):
        """List contacts in a specific category."""
        for name, info in self.contacts.items():
            if info['category'] == category:
                print(f"  {name}: {info['phone']}")

import json
import random
import time

class QuizGame:
    def __init__(self, questions_file="questions.json"):
        with open(questions_file, 'r') as f:
            self.all_questions = json.load(f)
        self.score = 0
        self.total = 0
        self.high_scores = self._load_high_scores()

    def _load_high_scores(self):
        try:
            with open("high_scores.json", 'r') as f:
                return json.load(f)
        except (FileNotFoundError, json.JSONDecodeError):
            return []

    def _save_high_score(self, name, score, total, time_taken):
        self.high_scores.append({
            'name': name,
            'score': score,
            'total': total,
            'percentage': round(score / total * 100) if total > 0 else 0,
            'time': round(time_taken, 1)
        })
        self.high_scores.sort(key=lambda x: x['percentage'], reverse=True)
        with open("high_scores.json", 'w') as f:
            json.dump(self.high_scores, f, indent=2)

    def ask_question(self, q):
        """Ask one question and return True if correct."""
        print(f"\n  {q['question']}")
        for i, option in enumerate(q['options']):
            print(f"    {i + 1}. {option}")

        try:
            answer = int(input("  Your answer (number): ")) - 1
        except ValueError:
            answer = -1

        if answer == q['answer']:
            print("  ✅ Correct!")
            return True
        else:
            correct = q['options'][q['answer']]
            print(f"  ❌ Wrong! The answer was: {correct}")
            return False

    def play(self, category):
        """Play a quiz in a given category."""
        if category not in self.all_questions:
            print(f"Category '{category}' not found!")
            return

        questions = self.all_questions[category][:]
        random.shuffle(questions)

        print(f"\n🧠 Quiz: {category.upper()} ({len(questions)} questions)")
        print("=" * 35)

        self.score = 0
        self.total = len(questions)
        start_time = time.time()

        for i, q in enumerate(questions, 1):
            print(f"\nQuestion {i}/{self.total}:")
            if self.ask_question(q):
                self.score += 1

        elapsed = time.time() - start_time
        pct = round(self.score / self.total * 100)

        print(f"\n{'=' * 35}")
        print(f"Score: {self.score}/{self.total} ({pct}%)")
        print(f"Time: {elapsed:.1f} seconds")

        if pct == 100:
            print("🏆 PERFECT SCORE!")
        elif pct >= 70:
            print("⭐ Great job!")
        else:
            print("📚 Keep studying!")

        name = input("\nYour name for the leaderboard: ").strip()
        if name:
            self._save_high_score(name, self.score, self.total, elapsed)

    def show_high_scores(self):
        print("\n🏆 HIGH SCORES:")
        for i, hs in enumerate(self.high_scores[:5], 1):
            print(f"  {i}. {hs['name']} - {hs['percentage']}% ({hs['score']}/{hs['total']}) in {hs['time']}s")

# Main

print("🧠 Quiz Game!")
print("Categories:", list(game.all_questions.keys()))
cat = input("Choose category: ").strip()
game.play(cat)
game.show_high_scores()

import requests
import datetime

class Dashboard:
    def __init__(self):
        self.sections = []

    def _safe_get(self, url, params=None, timeout=5):
        """Safe API call with error handling."""
        try:
            r = requests.get(url, params=params, timeout=timeout)
            r.raise_for_status()
            return r
        except requests.exceptions.RequestException:
            return None

    def get_weather(self, city="Tel+Aviv"):
        """Get weather from wttr.in."""
        r = self._safe_get(f"https://wttr.in/{city}?format=j1")
        if r:
            current = r.json()['current_condition'][0]
            desc = current['weatherDesc'][0]['value']
            return f"{city}: {current['temp_C']}°C, {desc}, Humidity: {current['humidity']}%"
        return "Weather unavailable"

    def get_joke(self):
        """Get random joke from JokeAPI."""
        r = self._safe_get("https://v2.jokeapi.dev/joke/Any", {'type': 'single'})
        if r:
            return r.json().get('joke', 'No joke found')
        return "Joke unavailable"

    def get_cat_fact(self):
        """Get random cat fact."""
        r = self._safe_get("https://catfact.ninja/fact")
        if r:
            return r.json()['fact']
        return "Cat fact unavailable"

    def get_number_fact(self):
        """Get fun fact about today's date."""
        day = datetime.datetime.now().day
        r = self._safe_get(f"http://numbersapi.com/{day}?json")
        if r:
            return r.json()['text']
        return "Number fact unavailable"

    def display(self, cities=None):
        """Display the full dashboard."""
        if cities is None:
            cities = ["Tel+Aviv"]

        now = datetime.datetime.now()
        width = 50

        print("\n" + "=" * width)
        print("  📊 DAILY INFO DASHBOARD")
        print(f"  {now.strftime('%Y-%m-%d %H:%M')}")
        print("=" * width)

        # Weather
        print("\n🌤️  WEATHER:")
        for city in cities:
            print(f"  {self.get_weather(city)}")

        # Joke
        print(f"\n🃏  JOKE OF THE MOMENT:")
        print(f"  {self.get_joke()}")

        # Cat fact
        print(f"\n🐱  CAT FACT:")
        print(f"  {self.get_cat_fact()}")

        # Number fact
        print(f"\n🔢  TODAY'S NUMBER ({now.day}):")
        print(f"  {self.get_number_fact()}")

        print("\n" + "=" * width)

# Run the dashboard!
dashboard = Dashboard()
dashboard.display(["Tel+Aviv", "London", "New+York"])
