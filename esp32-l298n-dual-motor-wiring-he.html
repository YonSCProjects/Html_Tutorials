מדריך חיווט ESP32 + L298N לשני מנועים

למכונית המחקר של מאיר (2 מנועים קדמיים)

פרויקט: מכונית נשלטת ESP32 עם מצלמה לצפייה בחיות בית

מנועים: 2 מנועי DC בחזית המכונית (ללא מנועים אחוריים)

מתח: סוללת Li-Po 3.7V

מה תלמד

בסוף המדריך הזה, יהיה לך:

  • ESP32 מופעל ועובד ✅
  • בקר מנועים L298N מחובר נכון ✅
  • שני מנועים מסתובבים באופן עצמאי ✅
  • הבנה איך החיווט עובד ✅

🔀 בחירה: איך להשתמש במדריך

אפשרות א': לעקוב צעד אחר צעד (מומלץ בפעם הראשונה)

אפשרות ב': להשתמש בתרשימים כהתייחסות תוך כדי חיווט עצמאי

שלב 1: הכנת לוח L298N

  1. הסתכל על לוח ה-L298N שלך
  2. מצא את כיסויי הג'מפר על פיני ENA ו-ENB
  3. הסר את שני כיסויי הג'מפר (נשלוט במהירות עם ESP32)
  4. שמור את הג'מפרים - אולי תצטרך אותם אחר כך

שלב 2: חיבור מנועים ל-L298N

מנוע שמאל (מנוע A):

  1. קח את שני החוטים של המנוע השמאלי
  2. שחרר את מסופי הבורג OUT1 ו-OUT2
  3. הכנס חוט אחד ל-OUT1, חוט אחד ל-OUT2
  4. הדק את הברגים היטב

מנוע ימין (מנוע B):

  1. קח את שני החוטים של המנוע הימני
  2. הכנס חוט אחד ל-OUT3, חוט אחד ל-OUT4
  3. הדק את הברגים היטב

שלב 5: חיבור מתח סוללה

⚠️ בדוק קוטביות שלוש פעמים לפני חיבור!
  1. אל תחבר סוללה עדיין - רק הכן את החוטים
  2. חוט אדום (+) של סוללה → מסוף +12V של L298N
  3. חוט שחור (-) של סוללה → מסוף GND של L298N

  • אדום ל-+12V, שחור ל-GND
  • אין חוטים רופפים
  • מנועים מאובטחים

שלב 6: הזנת ה-ESP32

אפשרות א': שימוש ביציאת 5V של L298N

  • חבר מסוף +5V של L298N → פין VIN של ESP32
  • זה מזין את ה-ESP32 מאותה סוללה

אפשרות ב': שימוש בכבל USB (יותר קל לבדיקות)

  • חבר כבל USB ל-ESP32
  • ESP32 מוזן מהמחשב, מנועים מהסוללה

לבדיקות, השתמש באפשרות ב' (USB) קודם!

קוד בדיקה

אחרי שהחיווט הושלם, העלה את הקוד הזה לבדיקת המנועים:

// ESP32 + L298N Dual Motor Test
// בדיקת מנועים כפולים - מכונית מחקר של מאיר

// פיני מנוע A (שמאל)
#define ENA 5   // בקרת מהירות PWM
#define IN1 16  // כיוון
#define IN2 17  // כיוון

// פיני מנוע B (ימין)
#define ENB 25  // בקרת מהירות PWM
#define IN3 27  // כיוון
#define IN4 26  // כיוון

void setup() {
  Serial.begin(115200);

  // הגדר את כל פיני המנוע כיציאות
  pinMode(ENA, OUTPUT);
  pinMode(IN1, OUTPUT);
  pinMode(IN2, OUTPUT);
  pinMode(ENB, OUTPUT);
  pinMode(IN3, OUTPUT);
  pinMode(IN4, OUTPUT);

  Serial.println("=== בדיקת מנועים מוכנה ===");
  Serial.println("פקודות:");
  Serial.println("  f = קדימה");
  Serial.println("  b = אחורה");
  Serial.println("  l = פנה שמאלה");
  Serial.println("  r = פנה ימינה");
  Serial.println("  s = עצור");
}

void loop() {
  if (Serial.available()) {
    char cmd = Serial.read();

    switch (cmd) {
      case 'f': // קדימה
        Serial.println("נוסע קדימה");
        motorA_forward();
        motorB_forward();
        break;

      case 'b': // אחורה
        Serial.println("נוסע אחורה");
        motorA_backward();
        motorB_backward();
        break;

      case 'l': // פנה שמאלה
        Serial.println("פונה שמאלה");
        motorA_backward();
        motorB_forward();
        break;

      case 'r': // פנה ימינה
        Serial.println("פונה ימינה");
        motorA_forward();
        motorB_backward();
        break;

      case 's': // עצור
        Serial.println("עוצר");
        motorA_stop();
        motorB_stop();
        break;
    }
  }
}

// פונקציות מנוע A (שמאל)
void motorA_forward() {
  digitalWrite(IN1, HIGH);
  digitalWrite(IN2, LOW);
  analogWrite(ENA, 200); // מהירות 0-255
}

void motorA_backward() {
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, HIGH);
  analogWrite(ENA, 200);
}

void motorA_stop() {
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, LOW);
  analogWrite(ENA, 0);
}

// פונקציות מנוע B (ימין)
void motorB_forward() {
  digitalWrite(IN3, HIGH);
  digitalWrite(IN4, LOW);
  analogWrite(ENB, 200);
}

void motorB_backward() {
  digitalWrite(IN3, LOW);
  digitalWrite(IN4, HIGH);
  analogWrite(ENB, 200);
}

void motorB_stop() {
  digitalWrite(IN3, LOW);
  digitalWrite(IN4, LOW);
  analogWrite(ENB, 0);
}

    • b = שני המנועים אחורה
    • l = פנה שמאלה
    • r = פנה ימינה
